import Foundation
import IndustrialKit
import SceneKit

public let <#Name#>_Module = PartModule(
    name: "<#Name#>",
    
    node: <#Name#>_Node
)

public var <#Name#>_Node: SCNNode
{
    guard let new_scene = SCNScene(named: "<#Name#>_Resources.scnassets/<#main_scene_name#>")
    else
    {
        return SCNNode()
    }
    
    guard let node = new_scene.rootNode.childNode(withName: "part", recursively: false)
    else
    {
        return SCNNode()
    }
    
    return node
}
